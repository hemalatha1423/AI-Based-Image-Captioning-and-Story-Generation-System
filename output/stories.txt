from transformers import AutoTokenizer, AutoModelForSeq2SeqLM
import torch

# Load FLAN-T5 properly
tokenizer = AutoTokenizer.from_pretrained("google/flan-t5-base")
model = AutoModelForSeq2SeqLM.from_pretrained("google/flan-t5-base")

device = "cuda" if torch.cuda.is_available() else "cpu"
model.to(device)

# ❌ words we don't want
banned_words = ["kill", "blood", "weapon", "attack", "war", "nazi"]

# 🔹 Clean repeated sentences
def clean_story(text):
    sentences = text.split(".")
    unique = []

    for s in sentences:
        s = s.strip()
        if s and s not in unique:
            unique.append(s)

    return ". ".join(unique) + "."

# 🔹 Check bad content
def is_bad(text):
    text = text.lower()
    return any(word in text for word in banned_words)

# 🔹 Fallback (always safe)
def fallback_story(caption):
    return (
        f"{caption.capitalize()} is captured in a calm and natural setting. "
        f"The surroundings create a peaceful and realistic atmosphere. "
        f"It reflects a simple and meaningful moment."
    )

# 🔹 MAIN FUNCTION
def generate_story(caption):
    caption = caption.strip().lower()

    prompt = f"""
Write a short and natural story based on this image:

{caption}

Rules:
- Exactly 3 sentences
- No repetition
- No violence
- No explanations
- Describe like a real-life moment
"""

    try:
        inputs = tokenizer(prompt, return_tensors="pt", truncation=True).to(device)

        outputs = model.generate(
            **inputs,
            max_length=80,
            num_beams=4,
            early_stopping=True
        )

        story = tokenizer.decode(outputs[0], skip_special_tokens=True)
        story = clean_story(story)

        # 🚨 safety check
        if is_bad(story) or len(story.split()) < 10:
            return fallback_story(caption)

        return story
